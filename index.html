    // Maximum subarray sum
    function maxSequence(arr) {
      if(arr.length===0) return 0;
      let maxSum = 0;
      let partialSum = 0;
    
      for (let item of arr) { 
        partialSum += item; 
        maxSum = Math.max(maxSum, partialSum);

      }
    
      return maxSum;
    }
